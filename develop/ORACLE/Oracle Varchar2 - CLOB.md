# Oracle Varchar2 -> CLOB

```sql
 ALTER TABLE ZIZZ.DC_SQLBOARD ADD (tmp_query CLOB);  
 UPDATE ZIZZ.DC_SQLBOARD SET tmp_query = query;
 ALTER TABLE ZIZZ.DC_SQLBOARD DROP COLUMN query;
 ALTER TABLE ZIZZ.DC_SQLBOARD RENAME COLUMN tmp_query TO query;
 SELECT ID,TITLE,DBMS_LOB.SUBSTR(QUERY, DBMS_LOB.GETLENGTH(QUERY), 1) as QUERY, VARLIST from DC_SQLBOARD_TEST;
```
